### 3. JS — renderRetentionTable()
```js
const RETENTION_BENCHMARKS = {
  W1:{good:40,bad:25}, W2:{good:25,bad:15}, W3:{good:20,bad:10},
  W4:{good:15,bad:8}, W8:{good:10,bad:5}, W12:{good:8,bad:3}
};
const RETENTION_WEEKS = ['W1','W2','W3','W4','W8','W12'];

function renderRetentionTable() {
  const audiences = loadAudiences();
  const saved = JSON.parse(localStorage.getItem('mp_retention_mc') || '{}');

  let html = '<thead><tr><th>Audience</th>';
  RETENTION_WEEKS.forEach(w => html += `<th>${w} %</th>`);
  html += '</tr></thead><tbody>';

  audiences.forEach(a => {
    html += `<tr><td class="kwt">${a.name}</td>`;
    RETENTION_WEEKS.forEach(w => {
      const val = saved[a.id]?.[w] ?? '';
      const cls = val !== '' ? getRetentionClass(parseFloat(val), w) : '';
      html += `<td><span class="kpi-cell ret-cell ${cls}" contenteditable="true"
        data-row="${a.id}" data-col="${w}"
        onblur="saveRetentionCell(this)">${val}</span></td>`;
    });
    html += '</tr>';
  });
  html += '</tbody>';
  document.getElementById('retention-table').innerHTML = html;
}

function getRetentionClass(val, week) {
  const b = RETENTION_BENCHMARKS[week];
  if (!b || isNaN(val)) return '';
  if (val >= b.good) return 'ret-good';
  if (val < b.bad) return 'ret-bad';
  return 'ret-mid';
}

function saveRetentionCell(el) {
  const data = JSON.parse(localStorage.getItem('mp_retention_mc') || '{}');
  if (!data[el.dataset.row]) data[el.dataset.row] = {};
  data[el.dataset.row][el.dataset.col] = el.textContent.trim();
  localStorage.setItem('mp_retention_mc', JSON.stringify(data));
  // Re-apply color class without full re-render
  const val = parseFloat(el.textContent.trim());
  el.className = `kpi-cell ret-cell ${getRetentionClass(val, el.dataset.col)}`;
}
```
