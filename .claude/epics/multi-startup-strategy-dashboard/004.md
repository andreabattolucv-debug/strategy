# T-04: KPI table editabile inline (localStorage)

## Obiettivo
Nuova sezione "KPI & Campaigns" condivisa, con tabella configurata da kpiSchema del progetto attivo.
Ogni cella è editabile inline con click. Dati persistono in localStorage per progetto.

### 3. JS — renderKpiTable()
```js
function renderKpiTable() {
  const schema = activeProject.kpiSchema;
  const rows = activeProject.boardType === 'audience'
    ? loadAudiences()
    : loadStrategies(); // esistente
  const key = `mp_kpi_${activeProject.id}`;
  const saved = JSON.parse(localStorage.getItem(key) || '{}');

  // Header row
  let html = '<thead><tr><th>Nome</th>';
  schema.forEach(k => html += `<th>${k}</th>`);
  html += '</tr></thead><tbody>';

  // Data rows
  rows.forEach(row => {
    html += `<tr><td class="kwt">${row.name || row.title}</td>`;
    schema.forEach(k => {
      const val = saved[row.id]?.[k] ?? '';
      html += `<td><span class="kpi-cell" contenteditable="true"
        data-row="${row.id}" data-col="${k}"
        onblur="saveKpiCell(this,'${activeProject.id}')">${val}</span></td>`;
    });
    html += '</tr>';
  });
  html += '</tbody>';
  document.getElementById('kpi-table').innerHTML = html;
}

function saveKpiCell(el, projectId) {
  const key = `mp_kpi_${projectId}`;
  const data = JSON.parse(localStorage.getItem(key) || '{}');
  if (!data[el.dataset.row]) data[el.dataset.row] = {};
  data[el.dataset.row][el.dataset.col] = el.textContent.trim();
  localStorage.setItem(key, JSON.stringify(data));
}
```
