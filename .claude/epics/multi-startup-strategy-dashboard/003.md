### 4. localStorage per audiences Mindcast
```js
function loadAudiences() {
  const saved = localStorage.getItem('mp_audiences_mc');
  return saved ? JSON.parse(saved) : MC_AUDIENCES_DEFAULT;
}
function saveAudiences(data) {
  localStorage.setItem('mp_audiences_mc', JSON.stringify(data));
}
```
