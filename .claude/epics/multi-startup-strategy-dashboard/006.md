# T-06: Funnel refactor — project + assignee + filtri

## Obiettivo
Aggiungere campi `project` e `assignee` ai todo esistenti del Funnel & Site Dev.
Aggiungere filtri (dropdown) sopra la lista. Dati separati in localStorage per progetto.
I todo esistenti di PN devono essere migrati automaticamente senza perdita dati.

## Lavoro da fare

### 1. Migrazione dati esistenti
```js
function migrateTodos() {
  const old = localStorage.getItem('mp_todos');
  if (old && !localStorage.getItem('mp_todos_migrated')) {
    const todos = JSON.parse(old);
    // Assegna project:'pn' a tutti i todo esistenti
    const migrated = todos.map(t => ({...t, project:'pn', assignee: t.assignee || 'team'}));
    localStorage.setItem('mp_todos_pn', JSON.stringify(migrated));
    localStorage.setItem('mp_todos_migrated', '1');
  }
}
```

### 2. Funzioni load/save aggiornate
```js
function loadTodos(projectId) {
  return JSON.parse(localStorage.getItem(`mp_todos_${projectId}`) || '[]');
}
function saveTodos(projectId, todos) {
  localStorage.setItem(`mp_todos_${projectId}`, JSON.stringify(todos));
}
```

### 5. Aggiornare renderFunnel() con filtri
```js
function renderFunnel() {
  let todos = loadTodos(activeProject.id);
  const assigneeFilter = document.getElementById('filter-assignee')?.value || 'all';
  const statusFilter = document.getElementById('filter-todo-status')?.value || 'all';

  if (assigneeFilter !== 'all') todos = todos.filter(t => t.assignee === assigneeFilter);
  if (statusFilter !== 'all') todos = todos.filter(t => t.status === statusFilter);
  // ... render esistente invariato, usa todos filtrati
}
```

### 6. Badge sidebar aggiornato
```js
function updateBadges() {
  const todos = loadTodos(activeProject.id);
  const open = todos.filter(t => t.status !== 'done').length;
  document.getElementById('nb-todo').textContent = open || '';
}
```
